## 🎓 Usage Examples

### Basic Usage
```rust
use xmt_cli::toroidal::*;

// Load or create ledger
let mut ledger = ToroidalLedger::load_or_create()?;

// Add node
let node = EnergyNode::new("New Node".to_string(), NodeType::NewEarthCity)
    .with_energy(369);
ledger.add_node(node);

// Distribute
ledger.distribute_with_algorithm(DistributionAlgorithm::GoldenRatio);

// Check status
println!("Energy: {}", ledger.total_energy());
println!("Optimal: {}", ledger.is_optimal());
```
